       - In the message stream loop, after existing handling, add stuck detection:
         ```typescript
         // Track tool calls for stuck detection
         if (stuckDetector && msg.type === 'assistant') {
           // Check content blocks for tool_use
           const content = msg.content as Array<Record<string, unknown>> | undefined;
           if (content) {
             for (const block of content) {
               if (block.type === 'tool_use') {
                 const toolName = block.name as string;
                 const toolInput = JSON.stringify(block.input ?? {});
                 if (stuckDetector.record(toolName, toolInput)) {
                   logger.stuck.error({ toolName, sessionId }, 'Stuck agent detected');
                   controller.abort();
                   return { success: false, sessionId, compactionCount, thresholdHit: false, costUsd: 0, resultSubtype: 'stuck', stuck: true };
                 }
               }
             }
           }
         }
         ```
